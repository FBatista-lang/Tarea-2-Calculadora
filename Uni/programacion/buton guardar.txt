private void bguardar1ActionPerformed(java.awt.event.ActionEvent evt) {                                          
        String cadena2, cadena3, cadena4, cadena5, cadena6, cadena7;
        cadena2 = nombre_user.getText();
        cadena3 = apellido_user.getText();
        cadena4 = correo.getText();
        cadena5 = telefono.getText();
        cadena6 = id_user.getText();
        cadena7 = contraseña_user.getText().toString();
        if (nombre_user.getText().equals("") || (apellido_user.getText().equals("")) || (correo.getText().equals("")) || (telefono.getText().equals("") || id_user.getText().equals("") || contraseña_user.getText().equals(""))) {
            javax.swing.JOptionPane.showMessageDialog(this, "Debes llenar todos los datos \n", "AVISO!!!!", javax.swing.JOptionPane.INFORMATION_MESSAGE);
            nombre_user.requestFocus();
    }                                         
    else {

            try {
                String URL = "jdbc:mysql://localhost:3306/users";
                String USER = "root";
                String CLAVE = "";
                Class.forName("com.mysql.jdbc.Driver");
                conectar = DriverManager.getConnection(URL, USER, CLAVE);
                if (conectar != null) {
                    System.out.println("Se ha establecido la conexion: " + "\n " + URL);
                }

                stmt = conectar.createStatement();
                stmt.executeUpdate("INSERT INTO registros Values ('" + 0 + "','" + cadena2 + "','" + cadena3 + "','" + cadena4 + "','" + cadena5 + "','" + cadena6 + "','" + cadena7 + "')");
                System.out.println("Se han guardado los datos...");

            } catch (ClassNotFoundException ex) {
                Logger.getLogger(registro_usuarios.class.getName()).log(Level.SEVERE, null, ex);
            } catch (SQLException ex) {
                Logger.getLogger(registro_usuarios.class.getName()).log(Level.SEVERE, null, ex);
            } finally {
                if (conectar != null) {
                    try {
                        conectar.close();
                        stmt.close();

                    } catch (Exception ex) {
                        System.out.println(ex.getMessage());

                    }

                }

            }
            javax.swing.JOptionPane.showMessageDialog(this, "Registro exitoso!!!! \n", "Aviso", javax.swing.JOptionPane.INFORMATION_MESSAGE);

        }
        this.nombre_user.setText("");
        this.apellido_user.setText("");
        this.contraseña_user.setText("");
        this.correo.setText("");
        this.id_user.setText("");
        this.telefono.setText("");
    }
